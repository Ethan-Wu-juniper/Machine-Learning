def ann_model():
    model = Sequential()
    model.add(Dense(20,input_dim=6213, activation='relu'))
    #model.add(Dropout(0.2))
    model.add(Dense(20,activation='softmax'))
    model.compile(loss='categorical_crossentropy', optimizer='adam', metrics=['accuracy'])
    return model
